Apps Script 편집기 → 기존 doGet 함수 맨 앞에 아래 블록 삽입 → 저장 → 재배포
if (e.parameter.action === 'getCsv') {
  var ss = SpreadsheetApp.openById(e.parameter.id);
  var sheets = ss.getSheets();
  var sheet = sheets.filter(function(s){
    return String(s.getSheetId()) === e.parameter.gid;
  })[0];
  if (!sheet) return ContentService
    .createTextOutput('').setMimeType(ContentService.MimeType.TEXT);
  var csv = sheet.getDataRange().getValues()
    .map(function(r){ return r.map(function(c){
      return '"' + String(c).replace(/"/g,'""') + '"';
    }).join(','); }).join('\n');
  return ContentService.createTextOutput(csv)
    .setMimeType(ContentService.MimeType.TEXT);
}
